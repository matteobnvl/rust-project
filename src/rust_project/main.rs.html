59    pub fn update(&mut self) {
60        for robot in &mut self.robots {
61            if robot.robot_type == robot::RobotType::Eclaireur {
62                robot::move_robot(robot, &mut self.map, self.width, self.height);
63                self.map_discovered
64                    .extend(robot.map_discovered.iter().map(|(x, y)| (*x, y.clone())));
65            }
66        }
67
68        // let resources_left = self.map_discovered
69        //     .values()
70        //     .filter(|t| matches!(t, map::Tile::SourceFound(qty) | map::Tile::CristalFound(qty) if *qty > 0))
71        //     .count();
72
73        let mut reserved_positions: std::collections::HashSet<(u16, u16)> = self
74            .robots
75            .iter()
76            .filter_map(|r| r.target_resource)
77            .map(|pos| (pos.0, pos.1))
78            .collect();
79
80        for robot in &mut self.robots {
81            robot::get_discovered_map(robot, &self.map_discovered);
82
83            if robot.robot_type == robot::RobotType::Collecteur {
84                if robot.target_resource.is_none() {
85                    for ((x, y), _tile) in self.map_discovered.clone() {
86                        match self.map[y as usize][x as usize] {
87                            map::Tile::Explored => {
88                                self.map_discovered.insert((x, y), map::Tile::Explored);
89                            }
90                            map::Tile::SourceFound(qty) | map::Tile::CristalFound(qty)
91                                if qty == 0 =>
92                            {
93                                self.map_discovered.insert((x, y), map::Tile::Explored);
94                            }
95                            _ => {}
96                        }
97                    }
98
99                    if let Some(new_target) = robot::find_nearest_resource(
100                        robot,
101                        &self.map_discovered,
102                        &reserved_positions,
103                    ) {
104                        robot.target_resource = Some(new_target);
105                        reserved_positions.insert((new_target.0, new_target.1));
106                    }
107                }
108
109                if let Some(_target) = robot.target_resource {
110                    let tx_base = self.tx_base.clone();
111                    let before = robot.target_resource;
112                    robot::collect_resources(
113                        robot,
114                        &mut self.map,
115                        self.width,
116                        self.height,
117                        &tx_base,
118                        &reserved_positions,
119                    );
120
121                    if let Some(target) = before
122                        && matches!(
123                            self.map[target.1 as usize][target.0 as usize],
124                            map::Tile::Explored
125                        )
126                    {
127                        self.map_discovered
128                            .insert((target.0, target.1), map::Tile::Explored);
129                    }
130                }
131            }
132        }
133
134        let base_center = (self.width / 2, self.height / 2);
135
136        for dy in -1..=1 {
137            for dx in -1..=1 {
138                let bx = (base_center.0 as i16 + dx) as usize;
139                let by = (base_center.1 as i16 + dy) as usize;
140                self.map[by][bx] = map::Tile::Base;
141            }
142        }
143    }
144}

222fn run(mut terminal: DefaultTerminal, game_state: &mut GameState, area: Size) -> Result<()> {
223    const TICK_RATE: Duration = Duration::from_millis(50);
224
225    let mut last_tick = Instant::now();
226    event::poll(Duration::from_millis(0)).map_err(SimulationError::Io)?;
227    tracing::info!("Crossterm configured");
228    loop {
229        if last_tick.elapsed() >= TICK_RATE {
230            game_state.update();
231        }
232
233        while let Ok(msg) = game_state.rx_broadcast.try_recv() {
234            if let base::BroadcastMessage::BaseStats { energy, crystals } = msg {
235                game_state.energy = energy;
236                game_state.crystals = crystals;
237                tracing::info!(
238                    "üèÜ Score mis √† jour : √©nergie = {}, cristaux = {}",
239                    energy,
240                    crystals
241                );
242            }
243        }
244
245        last_tick = Instant::now();
246
247        let timeout = TICK_RATE
248            .checked_sub(last_tick.elapsed())
249            .unwrap_or(Duration::from_millis(0));
250        if event::poll(timeout).map_err(SimulationError::Io)?
251            && let Event::Key(key_event) = event::read().map_err(SimulationError::Io)?
252            && key_event.code == KeyCode::Char(' ')
253        {
254            tracing::info!("Space key pressed, exiting game loop");
255            return Ok(());
256        }
257
258        terminal
259            .draw(|f| render_map_simple(f, game_state, area))
260            .map_err(SimulationError::Io)?;
261    }
262}

264fn render_map_simple(f: &mut Frame<'_>, game_state: &GameState, area: Size) {
265    let score_text = vec![Line::from(vec![
266        Span::styled("√ânergie: ", Style::default().fg(Color::Green)),
267        Span::styled(
268            game_state.energy.to_string(),
269            Style::default().fg(Color::White),
270        ),
271        Span::raw("   "),
272        Span::styled("Cristaux: ", Style::default().fg(Color::Magenta)),
273        Span::styled(
274            game_state.crystals.to_string(),
275            Style::default().fg(Color::White),
276        ),
277    ])];
278    let score_widget = Paragraph::new(score_text);
279    f.render_widget(score_widget, Rect::new(0, 0, area.width, 1));
280
281    let map_lines: Vec<Line> = game_state
282        .map
283        .iter()
284        .enumerate()
285        .take((game_state.height.saturating_sub(1)) as usize)
286        .map(|(y, row)| {
287            let spans: Vec<Span> = row
288                .iter()
289                .enumerate()
290                .take(game_state.width as usize)
291                .map(|(x, tile)| {
292                    let robot_here = game_state
293                        .robots
294                        .iter()
295                        .find(|r| r.position.0 == x as u16 && r.position.1 == y as u16);
296
297                    let (ch, color) = if let Some(robot) = robot_here {
298                        match robot.robot_type {
299                            robot::RobotType::Eclaireur => ('X', Color::Red),
300                            robot::RobotType::Collecteur => ('O', Color::Magenta),
301                        }
302                    } else {
303                        match tile {
304                            map::Tile::Wall => ('0', Color::LightCyan),
305                            map::Tile::Floor => (' ', Color::Reset),
306                            map::Tile::Source(_qty) => ('E', Color::Green),
307                            map::Tile::SourceFound(qty) => {
308                                if *qty > 0 {
309                                    ('E', Color::Blue)
310                                } else {
311                                    ('‚ñë', Color::Gray)
312                                }
313                            }
314                            map::Tile::Cristal(_qty) => ('C', Color::LightMagenta),
315                            map::Tile::CristalFound(qty) => {
316                                if *qty > 0 {
317                                    ('C', Color::Yellow)
318                                } else {
319                                    ('‚ñë', Color::Gray)
320                                }
321                            }
322                            map::Tile::Base => ('#', Color::LightGreen),
323                            // map::Tile::Eclaireur => ('X', Color::Red),
324                            // map::Tile::Collecteur => ('O', Color::Magenta),
325                            map::Tile::Explored => ('‚ñë', Color::Gray),
326                        }
327                    };
328                    Span::styled(ch.to_string(), Style::default().fg(color))
329                })
330                .collect();
331            Line::from(spans)
332        })
333        .collect();
334
335    let map_widget = Paragraph::new(map_lines);
336    f.render_widget(map_widget, Rect::new(0, 1, area.width, area.height));
