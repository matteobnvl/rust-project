22#[derive(Debug, thiserror::Error)]
23pub enum SimulationError {
24    Io(#[from] std::io::Error),
25}
26
27impl Display for SimulationError {
28    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
29        write!(f, "{:?}", self)
30    }
31}

33pub type Result<T> = std::result::Result<T, SimulationError>;

35#[tokio::main]
36async fn main() -> Result<()> {
37    let _guard = utils::configure_logger();
38    tracing::info!("Application started!");
39
40    // rng et channels setup
41    const REPEATED_SEED: [u8; 32] = [0; 32];
42    let _rng = StdRng::from_seed(REPEATED_SEED);
43    let (tx_base, rx_base) = mpsc::channel::<base::BaseMessage>(1024);
44    let (tx_broadcast, rx_broadcast) = broadcast::channel::<base::BroadcastMessage>(1024);
45
46    // base setup
47    let base = base::Base::new(tx_broadcast.clone());
48    let base_clone = base.clone();
49    tokio::spawn(async move {
50        base_clone.run(rx_base).await;
51    });
52
53    // terminal setup
54    let terminal = ratatui::init();
55    let area: Size = terminal.size().map_err(SimulationError::Io)?;
56
57    // map generation
58    let mut map = map::generate_map(area.width, area.height - 1)?;
59    let sources = map::generate_sources_rand(area.width, area.height - 1)?;
60    sources.iter().for_each(|(x, y, resource)| {
61        if let map::Tile::Floor = map[*y as usize][*x as usize] {
62            map[*y as usize][*x as usize] = resource.clone();
63        }
64    });
65
66    // base center generation
67    let start_x = (area.width / 2) - 1;
68    let start_y = (area.height / 2) - 1;
69    for y in start_y..start_y + 3 {
70        for x in start_x..start_x + 3 {
71            map[y as usize][x as usize] = map::Tile::Base;
72        }
73    }
74
75    // robots generation -- A REFACTO
76    let robot1 = robot::robots_eclaireur(area.width, area.height, (1, 0));
77    let robot2 = robot::robots_eclaireur(area.width, area.height, (0, 1));
78    let robot3 = robot::robots_collecteur(area.width, area.height);
79    let robot4 = robot::robots_collecteur(area.width, area.height);
80
81    // game configuration -- A REFACTO
82    tracing::info!("Map generated");
83    let mut game_state = GameState::new(
84        map,
85        area.width,
86        area.height,
87        vec![robot1, robot2, robot3, robot4],
88        base,
89        rx_broadcast,
90        tx_base.clone(),
91    );
92
93    tracing::info!("Game state initialized");
94
95    let res = run(terminal, &mut game_state, area);
96    tracing::info!("Game loop exited");
97    ratatui::restore();
98    res
99}

101fn run(mut terminal: DefaultTerminal, game_state: &mut GameState, area: Size) -> Result<()> {
102    const TICK_RATE: Duration = Duration::from_millis(50);
103
104    let mut last_tick = Instant::now();
105    event::poll(Duration::from_millis(0)).map_err(SimulationError::Io)?;
106    tracing::info!("Crossterm configured");
107    loop {
108        if last_tick.elapsed() >= TICK_RATE {
109            game_state.update();
110        }
111
112        while let Ok(msg) = game_state.rx_broadcast.try_recv() {
113            match msg {
114                base::BroadcastMessage::BaseStats { energy, crystals } => {
115                    game_state.energy = energy;
116                    game_state.crystals = crystals;
117                }
118            }
119        }
120
121        last_tick = Instant::now();
122
123        let timeout = TICK_RATE
124            .checked_sub(last_tick.elapsed())
125            .unwrap_or(Duration::from_millis(0));
126        if event::poll(timeout).map_err(SimulationError::Io)?
127            && let Event::Key(key_event) = event::read().map_err(SimulationError::Io)?
128            && key_event.code == KeyCode::Char(' ')
129        {
130            tracing::info!("Space key pressed, exiting game loop");
131            return Ok(());
132        }
133
134        terminal
135            .draw(|f| render_map_simple(f, game_state, area))
136            .map_err(SimulationError::Io)?;
137    }
138}

140fn render_map_simple(f: &mut Frame<'_>, game_state: &GameState, area: Size) {
141    let score_text = vec![Line::from(vec![
142        Span::styled("Énergie: ", Style::default().fg(Color::Green)),
143        Span::styled(
144            game_state.energy.to_string(),
145            Style::default().fg(Color::White),
146        ),
147        Span::raw("   "),
148        Span::styled("Cristaux: ", Style::default().fg(Color::Magenta)),
149        Span::styled(
150            game_state.crystals.to_string(),
151            Style::default().fg(Color::White),
152        ),
153    ])];
154    let score_widget = Paragraph::new(score_text);
155    f.render_widget(score_widget, Rect::new(0, 0, area.width, 1));
156
157    let map_lines: Vec<Line> = game_state
158        .map
159        .iter()
160        .enumerate()
161        .take((game_state.height.saturating_sub(1)) as usize)
162        .map(|(y, row)| {
163            let spans: Vec<Span> = row
164                .iter()
165                .enumerate()
166                .take(game_state.width as usize)
167                .map(|(x, tile)| {
168                    let robot_here = game_state
169                        .robots
170                        .iter()
171                        .find(|r| r.position.0 == x as u16 && r.position.1 == y as u16);
172
173                    let (ch, color) = if let Some(robot) = robot_here {
174                        match robot.robot_type {
175                            robot::RobotType::Eclaireur => ('X', Color::Red),
176                            robot::RobotType::Collecteur => ('O', Color::Magenta),
177                        }
178                    } else {
179                        match tile {
180                            map::Tile::Wall => ('0', Color::LightCyan),
181                            map::Tile::Floor => (' ', Color::Reset),
182                            map::Tile::Source(_qty) => ('E', Color::Green),
183                            map::Tile::SourceFound(qty) => {
184                                if *qty > 0 {
185                                    ('E', Color::Blue)
186                                } else {
187                                    ('░', Color::Gray)
188                                }
189                            }
190                            map::Tile::Cristal(_qty) => ('C', Color::LightMagenta),
191                            map::Tile::CristalFound(qty) => {
192                                if *qty > 0 {
193                                    ('C', Color::Yellow)
194                                } else {
195                                    ('░', Color::Gray)
196                                }
197                            }
198                            map::Tile::Base => ('#', Color::LightGreen),
199                            // map::Tile::Eclaireur => ('X', Color::Red),
200                            // map::Tile::Collecteur => ('O', Color::Magenta),
201                            map::Tile::Explored => ('░', Color::Gray),
202                        }
203                    };
204                    Span::styled(ch.to_string(), Style::default().fg(color))
205                })
206                .collect();
207            Line::from(spans)
208        })
209        .collect();
210
211    let map_widget = Paragraph::new(map_lines);
212    f.render_widget(map_widget, Rect::new(0, 1, area.width, area.height));
213}
