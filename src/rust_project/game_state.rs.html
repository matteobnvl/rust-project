1use crate::{base, map, robot};
2use std::collections::{HashMap, HashSet};
3use std::sync::{Arc, Mutex};
4use std::thread;
5use tokio::sync::mpsc;
6
7pub struct GameState {
8    pub(crate) map: Vec<Vec<map::Tile>>,
9    pub(crate) width: u16,
10    pub(crate) height: u16,
11    pub(crate) robots: Vec<robot::Robot>,
12    map_discovered: HashMap<(u16, u16), map::Tile>,
13    _base: base::SharedBase,
14    pub energy: u32,
15    pub crystals: u32,
16    pub rx_broadcast: tokio::sync::broadcast::Receiver<base::BroadcastMessage>,
17    pub tx_base: mpsc::Sender<base::BaseMessage>,
18    pub last_visited: HashMap<(u16, u16), usize>,
19    pub pending_resources: HashSet<(u16, u16)>,
20}
21
22impl GameState {
23    pub fn new(
24        map: Vec<Vec<map::Tile>>,
25        width: u16,
26        height: u16,
27        robots: Vec<robot::Robot>,
28        base: base::SharedBase,
29        rx_broadcast: tokio::sync::broadcast::Receiver<base::BroadcastMessage>,
30        tx_base: mpsc::Sender<base::BaseMessage>,
31    ) -> Self {
32        Self {
33            map,
34            width,
35            height,
36            robots,
37            map_discovered: HashMap::new(),
38            _base: base,
39            energy: 0,
40            crystals: 0,
41            rx_broadcast,
42            tx_base,
43            last_visited: HashMap::new(),
44            pending_resources: HashSet::new(),
45        }
46    }
47
48    pub fn update(&mut self) {
49        // Collecter les positions des éclaireurs
50        let eclaireur_positions: HashSet<(u16, u16)> = self
51            .robots
52            .iter()
53            .filter(|r| r.robot_type == robot::RobotType::Eclaireur)
54            .map(|r| (r.position.0, r.position.1))
55            .collect();
56
57        // Données partagées entre threads (avec Arc + Mutex)
58        let map_shared = Arc::new(Mutex::new(self.map.clone()));
59        let last_visited_shared = Arc::new(Mutex::new(self.last_visited.clone()));
60        let pending_shared = Arc::new(Mutex::new(self.pending_resources.clone()));
61
62        // Séparer éclaireurs et collecteurs
63        let mut eclaireurs = Vec::new();
64        let mut collecteurs = Vec::new();
65
66        for robot in self.robots.drain(..) {
67            if robot.robot_type == robot::RobotType::Eclaireur {
68                eclaireurs.push(robot);
69            } else {
70                collecteurs.push(robot);
71            }
72        }
73
74        // LANCER LES ÉCLAIREURS EN PARALLÈLE
75        let handles: Vec<_> = eclaireurs
76            .into_iter()
77            .enumerate()
78            .map(|(robot_id, mut robot)| {
79                let map_clone = Arc::clone(&map_shared);
80                let last_visited_clone = Arc::clone(&last_visited_shared);
81                let pending_clone = Arc::clone(&pending_shared);
82                let eclaireur_pos = eclaireur_positions.clone();
83                let width = self.width;
84                let height = self.height;
85
86                thread::spawn(move || {
87                    let other_positions: HashSet<(u16, u16)> = eclaireur_pos
88                        .iter()
89                        .filter(|&&pos| pos != (robot.position.0, robot.position.1))
90                        .copied()
91                        .collect();
92
93                    // Mettre à jour last_visited
94                    {
95                        let mut lv = last_visited_clone.lock().unwrap();
96                        lv.insert((robot.position.0, robot.position.1), robot_id);
97                    }
98
99                    // Appeler move_robot avec les locks
100                    {
101                        let mut map = map_clone.lock().unwrap();
102                        let lv = last_visited_clone.lock().unwrap();
103                        let mut pending = pending_clone.lock().unwrap();
104
105                        robot::move_robot(
106                            &mut robot,
107                            &mut map,
108                            width,
109                            height - 1,
110                            &other_positions,
111                            &lv,
112                            robot_id,
113                            &mut pending,
114                        );
115                    }
116
117                    robot
118                })
119            })
120            .collect();
121
122        // Attendre que tous les threads se terminent
123        let mut eclaireurs: Vec<_> = handles
124            .into_iter()
125            .map(|h| h.join().expect("Thread éclaireur a paniqué"))
126            .collect();
127
128        // Récupérer les données partagées
129        self.map = Arc::try_unwrap(map_shared)
130            .expect("Arc still has references")
131            .into_inner()
132            .unwrap();
133        self.last_visited = Arc::try_unwrap(last_visited_shared)
134            .expect("Arc still has references")
135            .into_inner()
136            .unwrap();
137        self.pending_resources = Arc::try_unwrap(pending_shared)
138            .expect("Arc still has references")
139            .into_inner()
140            .unwrap();
141
142        // Mettre à jour map_discovered avec les découvertes de chaque éclaireur
143        for robot in &eclaireurs {
144            self.map_discovered
145                .extend(robot.map_discovered.iter().map(|(x, y)| (*x, y.clone())));
146        }
147
148        // Remettre les robots dans la liste
149        self.robots.append(&mut eclaireurs);
150        self.robots.append(&mut collecteurs);
151
152        // ⭐ COLLECTEURS (séquentiel, pas besoin de paralléliser)
153        let mut reserved_positions: HashSet<(u16, u16)> = self
154            .robots
155            .iter()
156            .filter(|r| r.robot_type == robot::RobotType::Collecteur)
157            .filter_map(|r| r.target_resource)
158            .map(|pos| (pos.0, pos.1))
159            .collect();
160
161        for robot in &mut self.robots {
162            robot::get_discovered_map(robot, &self.map_discovered);
163
164            if robot.robot_type == robot::RobotType::Collecteur {
165                if robot.target_resource.is_none() {
166                    for ((x, y), _tile) in self.map_discovered.clone() {
167                        match self.map[y as usize][x as usize] {
168                            map::Tile::Explored => {
169                                self.map_discovered.insert((x, y), map::Tile::Explored);
170                            }
171                            map::Tile::SourceFound(qty) | map::Tile::CristalFound(qty)
172                                if qty == 0 =>
173                            {
174                                self.map_discovered.insert((x, y), map::Tile::Explored);
175                            }
176                            _ => {}
177                        }
178                    }
179                    if let Some(new_target) = robot::find_nearest_resource(
180                        robot,
181                        &self.map_discovered,
182                        &reserved_positions,
183                    ) {
184                        robot.target_resource = Some(new_target);
185                        reserved_positions.insert((new_target.0, new_target.1));
186                    }
187                }
188
189                if let Some(_target) = robot.target_resource {
190                    let tx_base = self.tx_base.clone();
191                    let before = robot.target_resource;
192                    robot::collect_resources(
193                        robot,
194                        &mut self.map,
195                        self.width,
196                        self.height,
197                        &tx_base,
198                        &reserved_positions,
199                    );
200
201                    if let Some(target) = before
202                        && matches!(
203                            self.map[target.1 as usize][target.0 as usize],
204                            map::Tile::Explored
205                        )
206                    {
207                        self.map_discovered
208                            .insert((target.0, target.1), map::Tile::Explored);
209                    }
210                }
211            }
212        }
213
214        // Redessiner la base
215        let base_center = (self.width / 2, self.height / 2);
216        for dy in -1..=1 {
217            for dx in -1..=1 {
218                let bx = (base_center.0 as i16 + dx) as usize;
219                let by = (base_center.1 as i16 + dy) as usize;
220                self.map[by][bx] = map::Tile::Base;
221            }
222        }
223    }
224}
